## Dev Notes

### Extração de ga_client_id

```typescript
function getClientId(measurementId?: string): string {
  const gaCookie = readCookie('_ga')
  if (gaCookie) {
    const parts = gaCookie.split('.')
    if (parts.length >= 4) return `${parts[2]}.${parts[3]}`
  }
  // Fallback: tentar localStorage
  const stored = localStorage.getItem('nx_ga4_cid')
  if (stored) return stored
  // Gerar novo
  const generated = `${Math.random().toString(36).substring(2)}.${Date.now()}`
  try { localStorage.setItem('nx_ga4_cid', generated) } catch {}
  return generated
}
```

### getSessionData

```typescript
function getSessionData(measurementId: string): { session_id: string; session_count: string } {
  const key = measurementId.replace('G-', '')
  const cookie = readCookie(`_ga_${key}`)
  if (cookie) {
    const parts = cookie.split('.')
    if (parts.length >= 4) return { session_id: parts[2], session_count: parts[3] }
  }
  return { session_id: '', session_count: '' }
}
```

### send_page_view: false

O `page_view` é disparado manualmente após carregamento do script via `onload`, para garantir que o `page_title` e `page_location` corretos estejam disponíveis:

```typescript
script.onload = () => {
  window.gtag('event', 'page_view', {
    page_location: window.location.href,
    page_title: document.title,
    page_referrer: document.referrer,
  })
}
```
