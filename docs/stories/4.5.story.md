## Story

Como landing page externa (WordPress, Elementor, etc.),
Quero um script que capture eventos de uma LP com datalayer manual,
Para que `PageView`, `Lead` e eventos customizados configuráveis sejam disparados ao worker e plataformas, e links de checkout externos sejam decorados com sinais para transfer cross-domain.

## Dev Notes

### Config triggers

```typescript
interface EventTrigger {
  triggerType: 'pageload' | 'click' | 'form_submit' | 'scroll' | 'time_on_page'
  eventName: EventName
  selector?: string          // CSS selector para click/form
  buttonText?: string        // texto do botão para click
  scrollDepth?: number       // percentual (0-100) para scroll
  timeSeconds?: number       // segundos para time_on_page
  customData?: CustomData
}
```

### TriggerEngine

```typescript
function initTriggerEngine(triggers: EventTrigger[], track: (name: EventName, data?: CustomData) => void): void {
  triggers.forEach(trigger => {
    switch (trigger.triggerType) {
      case 'pageload':
        track(trigger.eventName, trigger.customData)
        break

      case 'click':
        document.addEventListener('click', (e) => {
          const el = (e.target as Element).closest(trigger.selector || '*')
          if (!el) return
          if (trigger.buttonText) {
            const text = (el as HTMLElement).textContent?.trim().toLowerCase()
            if (!text?.includes(trigger.buttonText.toLowerCase())) return
          }
          track(trigger.eventName, trigger.customData)
        }, true)
        break

      case 'scroll':
        if (!trigger.scrollDepth) break
        let scrollFired = false
        window.addEventListener('scroll', () => {
          if (scrollFired) return
          const total = document.documentElement.scrollHeight || 1
          const pct = (window.scrollY + window.innerHeight) / total * 100
          if (pct >= trigger.scrollDepth!) {
            scrollFired = true
            track(trigger.eventName, trigger.customData)
          }
        }, { passive: true })
        break

      case 'time_on_page':
        if (trigger.timeSeconds && trigger.timeSeconds > 0) {
          setTimeout(() => track(trigger.eventName, trigger.customData), trigger.timeSeconds * 1000)
        }
        break

      case 'form_submit':
        document.addEventListener('submit', (e) => {
          const form = e.target as HTMLFormElement
          if (trigger.selector && !form.matches(trigger.selector)) return
          track(trigger.eventName, trigger.customData)
        })
        break
    }
  })
}
```

### Captura de lead via DOM

```typescript
let leadFired = false

function captureLeadFromDom(): void {
  if (leadFired) return
  const emailEl = document.querySelector<HTMLInputElement>('input[type="email"], input[name*="email"]')
  const phoneEl = document.querySelector<HTMLInputElement>('input[type="tel"], input[name*="phone"], input[name*="cellphone"]')
  const email = emailEl?.value?.includes('@') ? emailEl.value : null
  const phone = phoneEl?.value?.length > 8 ? phoneEl.value : null
  if (email || phone) {
    leadFired = true
    track('Lead', {}, { email: email ?? undefined, phone: phone ?? undefined })
  }
}

document.addEventListener('focusout', (e) => {
  const el = e.target as HTMLElement
  if (el instanceof HTMLInputElement &&
      (el.type === 'email' || el.type === 'tel' || el.name?.includes('email') || el.name?.includes('phone'))) {
    captureLeadFromDom()
  }
}, true)
```
