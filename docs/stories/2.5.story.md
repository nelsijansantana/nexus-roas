### DecoraГ§ГЈo de link

```typescript
const TRANSFER_PARAMS = [
  'utm_source', 'utm_medium', 'utm_campaign', 'utm_term', 'utm_content',
  'utm_id', 'fbclid', 'gclid', 'gbraid', 'wbraid', 'ttclid', 'msclkid',
]

function decorateLink(a: HTMLAnchorElement, nxUser: string, utms: Record<string, string>, signals: SignalMap): void {
  try {
    const url = new URL(a.href)
    if (!isCheckoutDomain(url.hostname)) return

    let modified = false
    TRANSFER_PARAMS.forEach(param => {
      const val = utms[param] || (signals as any)[param]
      if (val && !url.searchParams.has(param)) {
        url.searchParams.set(param, val)
        modified = true
      }
    })
    if (nxUser && !url.searchParams.has('src')) {
      url.searchParams.set('src', nxUser)
      url.searchParams.set('sck', nxUser)
      modified = true
    }
    if (modified) a.href = url.toString()
  } catch {}
}
```

### MutationObserver

```typescript
function startObserver(decorateFn: () => void): void {
  if (!window.MutationObserver) return
  const observer = new MutationObserver(mutations => {
    mutations.forEach(m => { if (m.addedNodes.length) decorateFn() })
  })
  const start = () => observer.observe(document.body, { childList: true, subtree: true })
  document.body ? start() : document.addEventListener('DOMContentLoaded', start)
}
```
